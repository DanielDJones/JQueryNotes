JQuery - Vanilla.js Rosetta Stone

This is brief document outlining some of the differences between vanilla JS and JQuery. To make it easier to translate between the two.

Please not that some code is taken from other sources and is just for untested reference.

Getting elements on a page. Getting a notice button for example.

JQuery selections are not live, so if you select all the .drawers for example then add another to the document then the new one will not be selected.

Vanilla js

                    
//ID ONLY

document.getElementById("notice-btn"); 


//Query Selector

document.querySelector("#notice-btn");

JQeuery

                    
$( "#notice-btn" );

Vanilla - JQuery Conversion

                    
//GET ELEMENT

var target = document.getElementById( "target" );


//WRAP TARGET INTO A JQUERY OBJ

$( target )

//USE
$( target ).html("New Inner HTML")

Vanilla js multiple

                    
//NOTICE THE ALL PART OF THE FUNCTION

document.querySelectorAll('.drawer');

JQeuery - multiple 

                    
// DUE TO IT BEING A CLASS IT SELECTS MULTIPLE

$( ".drawers" );

In the each function you can use the this keyword in order to get the element.
Each can also be used similarly to PHP's foreach($tests as $test){}

Vanilla js -Each on page 

                    
var SelectedItems = document.querySelectorAll('.drawer');
for (var i = 0, len = SelectedItems.length; i < len; i++) {
    //CODE HERE
}

JQeuery - Each on page 

                    
$( ".drawer" ).each(function() {
    $( this ).examplefunction()    
});

JQuery - For Each Without Select 

                    
//USED FOR PLAIN OBJECTS, ARRAYS, ARRAY-LIKE OBJECTS. DONT USE THIS WITH JQUERY COLLECTIONS

var obj = {
    foo: 1,
    bar: 2
}

$.each( obj, function( key, value ) {
    sum += value;
});
 
console.log( sum ); // 3

JQeuery - Where Each isnt needed 

                    
//YOU DONT NEED AN EACH WITH SOME FUNCTIONS AS THEY WORK ON MULTIPLE ELEMENTS IN ADDITION TO JUST A SINGLE ELEMENT

$ ( ".drawer" ).hide();

Event listeners can be done 2 different ways, the preferred way is to use the .on function as it uses less memory and can be used with dynamically added elements.

Vanilla js

                    
document.getElementById("update-btn").addEventListener("click", function() {
    // CODE TO PERFORM HERE
});

JQeuery

                    
//NO PARAMETER CALL THE HIDEELEMENT FUNCTION ON BUTTON CLICK

$( document ).on("click", "#notice-btn", HideElement)


//WITH PARAMETERS

$( document ).on("click", "#notice-btn", {ElementID: "#notice-banner"}, HideElement)

JQeuery - For Each + Event Listeners 

                    
//WILL CALL THE TOGGLEDRAWER FOR EACH BUTTON CLICK WITH THE CLASS .drawer-header

$( ".drawer-header" ).each(function() {
    $( document ).on("click", "#" + this.id , ToggleDrawer)
});

$( "#drawer" ).hide()

Calling an API with AJAX is pretty easy, these examples do not have built in error checking however. There are a few different functions but .ajax() gives the most flexibility.

JQuery -API Call With Errors

                    
$.ajax({
    //REQUEST
    method: "GET",
    url: "https://jsonplaceholder.typicode.com/users",
    dataType: "json",
    success: function (data, text) {

        //WILL RUN ON SUCCESS
        
        //PUT CODE HERE
          
    },
    error: function (request, status, error) {

        //LOG ERROR INFO

        console.log(request)
        console.log(status)
        console.log(error)
    }
})

JQuery - API Call (AJAX) - No Error handling

                    
//BASIC AJAX CALL

   $.ajax({
       //REQUEST

       method: "GET", //GET IS DEFAULT
       url: "https://jsonplaceholder.typicode.com/users", //PLACEHOLDER API
       dataType: "json" //EXPECTING JSON DATA

   }).done(function(data){ //WHEN THE CALL IS DONE
       
        //RESPONSE

       $.each(data, function(i, user){
            //CALL FUNCTION
            AddUser(this)
       })

   });

Vanilla JS adjusting CSS

                    
//ONE STYLE PROP AT A TIME
var box = document.querySelector(".vanilla");
box.style.color = "#000";
box.style.backgroundColor = "#ffff00";

JQuery adjusting CSS

                    
// YOU CAN ADJUST MORE THAN 1
// THIS ADJUSTS ALL INSTANCES OF THE .VANILLA CLASS

$(".vanilla").css({
  "color": "#000",
  "background-color": "#ffff00"
});

Vanilla

                    
//SELECT THE FIRST ELEMENT WITH THE CLASS DRAWER

var Drawer = document.querySelector(".drawer");

//ADD

Drawer.classList.add("red");

//REMOVE

Drawer.classList.remove("red");

//TOGGLE

Drawer.classList.toggle("red");

JQuery

                  
//THESE EFFECT ALL ELEMENTS WITH THE CLASS

//ADD

$( ".drawer" ).addClass("red")

//Remove

$( ".drawer" ).removeClass("red")

//Toggle

$( ".drawer" ).toggleClass("red")

Vanilla JS

                    
document.getElementById('example').setAttribute("href", "https://www.website.example")

JQuery

                    
$( "#example" ).attr( "href", "https://www.website.example" );

//YOU CAN CHANGE MULTIPLE AT A TIME

$( "#example" ).attr({
    hreflang: "en",
    href: "https://www.website.example"
});

Do Things when the page is fully loaded

Vanilla JS

                    

var ready = (callback) => {
  if (document.readyState != "loading") callback();
  else document.addEventListener("DOMContentLoaded", callback);
}

ready(() => {

    //CODE HERE 

});

JQuery

                    
$(document).ready(function() {

    //CODE HERE 

});

JQuery - Other Ways

                    
//SHORHAND
$(function() {
    console.log( "ready!" );
});



//SEPERATE FN

function ReadyFN( jQuery ) {

    //CODE HERE

}
    
    
    
//THEN CALL IT

//CALL TYPE 1
$( document ).ready( ReadyFN );

//CALL TYPE 2
$( window ).on( "load", ReadyFN);
